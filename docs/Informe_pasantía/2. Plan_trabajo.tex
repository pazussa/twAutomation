\section{Plan de trabajo avalado}
\label{sec:methodology}




\subsection{Actividades}

\subsubsection{Preparación y configuración inicial}
\begin{itemize}
  \item Recopilar criterios de aceptación por HU.
  \item Reescribir criterios de aceptación iniciales en Gherkin
  \item Configurar repo TS + Cucumber + Playwright.
  \item Login persistente a WhatsApp Web (state.json) y chat Twilio o Whatsapp Business.
  \item Definir convenciones: tags por HU/Fase, nombres de steps, estructura de\\
  carpetas, formato de reporte.
\end{itemize}

\subsubsection{Base de automatización}
\begin{itemize}
  \item POM WhatsApp: abrir chat, enviar mensaje, esperar respuesta, confirmar textos,\\
  manejar multi-turno.
  \item Wait helpers: esperas activas por “pregunta del bot”, “confirma”, “listado”.
  \item Message parsers: utilidades para detectar prompts del bot y listas/tablas en texto.
  \item Modelo de datos mínimo: crear/limpiar cultivos y productos de prueba.
\end{itemize}

\subsubsection{Automatización de Fase 1}
\begin{itemize}
  \item HU‐002: Escenarios de usuario válido/ inválido y sesión expirada.
  \item HU‐006: Listado y filtrado de cultivos.
  \item HU‐007 a HU‐019: Listado de variedades, fertilizantes, fitosanitarios y otros\\
  recursos.
  \item Refactorización de pasos comunes (inicio/cierre de sesión, navegación).
  \item Registro de incidentes detectados durante la ejecución.
\end{itemize}

\subsubsection{Automatización de Fase 2}
\begin{itemize}
  \item HU‐020/021/022: Creación de productos (cultivo, fertilizante, químico) con\\
  validación de campos.
  \item HU‐003: Listar productos por fabricante
  \item HU‐004: Asignación de precio de producto y verificación de persistencia.
  \item HU‐005: Búsqueda por materia activa (coincidencias exactas y parciales).
  \item Reutilización de page objects y steps; generación de reportes básicos.
\end{itemize}

\subsubsection{Automatización de Fase 3}
\begin{itemize}
  \item HU‐023: Creación de campañas (manejo de datos y fechas).
  \item HU‐024: Consulta de historial de campañas (filtrado, orden, detalle).
  \item HU‐025: Planificación de trabajos a campos, confirmando estados.
  \item Integración parcial de flujos end‐to‐end combinando fases anteriores.
\end{itemize}

\subsubsection{Revisión, documentación y entrega}
\begin{itemize}
  \item Consolidar scripts y pasos en un README final con instrucciones detalladas.
  \item Especificar dependencias y cómo añadir nuevas pruebas.
  \item Redactar informe de cobertura, hallazgos y recomendaciones futuras para abordar\\
  CI/CD.
  \item Elaborar un informe de retrospectiva con aprendizajes y sugerencias.
\end{itemize}
